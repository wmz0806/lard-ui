### search header
<Playground>
  <SearchHeader 
    leftBtnConfig={{
      icon: 'fanhui',
      onClick: () => console.log('点击返回')
    }}
    searchConfig={{
      onChange: (e, value) => console.log('change', value),
      onChangeDelay: 500,
      onBlur: (e, val) => console.log('blur', val),
      onFocus: (e, val) => console.log('focus', val),
      onCancel: (e, val) => console.log('cancel', val),
      onClick: (e, val) => console.log('click', val),
      onKeyDown: (e, val) => console.log('keyDown', val),
      placeholder: 'placeholder',
      searchValue: '默认value'
    }}
    rightBtnConfig={{
      icon: 'cha',
      onClick: () => {}
    }}
    ref={r => window.SearchHeader = r}
  />
  <div onClick={() => window.SearchHeader.setActive(true)}>点击激活</div>
  <div onClick={() => window.SearchHeader.setActive(false)}>点击不激活</div>
</Playground>
