### 队列演示

<Playground>
 <Component initialState={{ images: [] }}>
    {component => {
      return (
        <Queue 
              max={3}
              images={component.state.images}
              delCallback={(index, item) => {
                  component.state.images.splice(index, 1);
                  component.setState({});
              }}
              uploadProps={{
                onComplete: (result, loadingCloseFun) => {
                 component.state.images.push(result.base64);
                 component.setState({});
                 loadingCloseFun();  //关闭loading
                }
              }}
             />
      );
    }}
  </Component>
  
</Playground>
